# ============================================
# 复制以下内容到 backend/.env 文件中
# ============================================

# 环境配置
NODE_ENV=development
PORT=5000

# ⭐ 本地PostgreSQL数据库连接（重要！）
# 格式：postgresql://用户名:密码@localhost:5432/数据库名
# 请将 YOUR_PASSWORD 替换为你的PostgreSQL密码
DATABASE_URL=postgresql://postgres:YOUR_PASSWORD@localhost:5432/vr_game_platform

# JWT密钥
JWT_SECRET=local-dev-secret-key-change-in-production
JWT_REFRESH_SECRET=local-dev-refresh-key-change-in-production
JWT_EXPIRES_IN=15m
JWT_REFRESH_EXPIRES_IN=7d

# 前端URL
FRONTEND_URL=http://localhost:3000

# Redis配置（本地开发暂时不需要）
REDIS_HOST=localhost
REDIS_PORT=6379
REDIS_PASSWORD=

# ============================================
# 使用步骤：
# ============================================
# 1. 在pgAdmin中创建数据库 vr_game_platform
# 2. 将上面内容复制到 backend/.env
# 3. 修改 YOUR_PASSWORD 为你的PostgreSQL密码
# 4. 运行命令：npm run db:fresh
# ============================================
